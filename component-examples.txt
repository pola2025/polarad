  id,
  type = 'text',
  placeholder,
  value,
  onChange,
  error,
  helperText,
  required = false
}) {
  return (
    <div className="input-wrapper">
      <label
        htmlFor={id}
        className={`input-label ${required ? 'input-label-required' : ''}`}
      >
        {label}
      </label>

      <input
        type={type}
        id={id}
        value={value}
        onChange={onChange}
        placeholder={placeholder}
        className={`input-field ${error ? 'error' : ''}`}
        aria-invalid={!!error}
        aria-describedby={error ? `${id}-error` : helperText ? `${id}-helper` : undefined}
      />

      {error && (
        <span id={`${id}-error`} className="input-error">
          {error}
        </span>
      )}

      {helperText && !error && (
        <span id={`${id}-helper`} className="input-helper">
          {helperText}
        </span>
      )}
    </div>
  );
}
